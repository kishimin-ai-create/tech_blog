**生成結果**:
- ✅ 11 個のストーリーファイル
- ✅ 95+ ストーリーバリアント
- ✅ 9 個の API エンドポイント完全モック
- ✅ Storybook ビルド完了（1.32 秒）
- ✅ 全ストーリーが TypeScript 型安全

## API エンドポイントと MSW モッキング

### エンドポイント一覧（9 個）

```typescript
// ==================== Apps API ====================

// 1. GET /api/v1/apps - アプリ一覧取得
http.get('/api/v1/apps', () =>
  HttpResponse.json({
    success: true,
    data: [
      { id: 'app-1', name: 'App 1', createdAt: '2026-04-01T00:00:00Z' },
      { id: 'app-2', name: 'App 2', createdAt: '2026-04-02T00:00:00Z' },
    ],
  }),
)

// 2. POST /api/v1/apps - アプリ新規作成
http.post('/api/v1/apps', async ({ request }) => {
  const { name } = await request.json()
  
  // 409 Conflict: 名前重複
  if (name === 'Duplicate Name') {
    return HttpResponse.json(
      {
        success: false,
        error: { code: 'CONFLICT', message: 'App name already exists' },
      },
      { status: 409 },
    )
  }
  
  // 201 Created: 成功
  return HttpResponse.json(
    {
      success: true,
      data: { id: 'app-new', name, createdAt: new Date().toISOString() },
    },
    { status: 201 },
  )
})

// 3. GET /api/v1/apps/:appId - アプリ詳細取得
http.get('/api/v1/apps/:appId', ({ params }) => {
  const { appId } = params
  
  if (appId === 'not-found') {
    return HttpResponse.json(
      { success: false, error: { code: 'NOT_FOUND' } },
      { status: 404 },
    )
  }
  
  return HttpResponse.json({
    success: true,
    data: { id: appId, name: 'Test App', createdAt: '2026-04-01T00:00:00Z' },
  })
})

// 4. PUT /api/v1/apps/:appId - アプリ更新
http.put('/api/v1/apps/:appId', async ({ request }) => {
  const { name } = await request.json()
  
  if (name.includes('conflict')) {
    return HttpResponse.json(
      { success: false, error: { code: 'CONFLICT' } },
      { status: 409 },
    )
  }
  
  return HttpResponse.json({
    success: true,
    data: { id: 'app-1', name, updatedAt: new Date().toISOString() },
  })
})

// 5. DELETE /api/v1/apps/:appId - アプリ削除
http.delete('/api/v1/apps/:appId', () =>
  HttpResponse.json({ success: true, data: null }, { status: 204 }),
)

// ==================== Todos API ====================

// 6. GET /api/v1/apps/:appId/todos - Todo 一覧取得
http.get('/api/v1/apps/:appId/todos', () =>
  HttpResponse.json({
    success: true,
    data: [
      { id: 'todo-1', title: 'Task 1', completed: false, appId: 'app-1' },
      { id: 'todo-2', title: 'Task 2', completed: true, appId: 'app-1' },
    ],
  }),
)

// 7. POST /api/v1/apps/:appId/todos - Todo 新規作成
http.post('/api/v1/apps/:appId/todos', async ({ request }) => {
  const { title } = await request.json()
  
  if (title.length > 200) {
    return HttpResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR' } },
      { status: 422 },
    )
  }
  
  return HttpResponse.json(
    {
      success: true,
      data: {
        id: 'todo-new',
        title,
        completed: false,
        appId: 'app-1',
        createdAt: new Date().toISOString(),
      },
    },
    { status: 201 },
  )
})

// 8. PUT /api/v1/apps/:appId/todos/:todoId - Todo 更新
http.put('/api/v1/apps/:appId/todos/:todoId', async ({ request }) => {
  const { title, completed } = await request.json()
  
  return HttpResponse.json({
    success: true,
    data: { id: 'todo-1', title, completed, appId: 'app-1' },
  })
})

// 9. DELETE /api/v1/apps/:appId/todos/:todoId - Todo 削除
http.delete('/api/v1/apps/:appId/todos/:todoId', () =>
  HttpResponse.json({ success: true, data: null }, { status: 204 }),
)
```
